Quiz Lists

1) What is the output of the following script?


let firstName = null;
let lastName = null;
let nickName = "Supercoder";
console.log(firstName ?? lastName ?? nickName ?? "Anonymous");

a) null
b) "Anonymous"
c) undefined
d) "Supercoder"


Answer : d

2) What is the output of the following script?


let sum = (a, b) => {
 let result = a + b;
 return result;
}
console.log(sum(1,2));

a) 3
b) 5
c) 2
d) 6


Answer : a

3) What is the output of the following script?


let multiply = (a, b) => {
 let result = a * b;
 return result;
}
console.log(multiply(4,5));

a) 19
b) 20
c) 24
d) 18


Answer : b

4) What is the output of the following script?


let firstName = null;
let lastName = null;
let nickName = "Supercoder";
console.log(firstName || lastName || nickName || "Anonymous");

a) null
b) "Anonymous"
c) undefined
d) "Supercoder"


Answer : d

5) What is the output of the following script?


let height = null;
let width = null;
let area = (height ?? 100) * (width ?? 50);
console.log(area);

a) null
b) 5000
c) undefined
d) 1500


Answer : b

6) What is the output of the following script?


let x = (1 && 2) ?? 3;
console.log(x);

a) 1
b) 3
c) 2
d) null


Answer : c
